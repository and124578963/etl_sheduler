 function makeCustom () {
	if (!grafanaBootData.user.isGrafanaAdmin){
		var list_header = document.querySelectorAll('.css-11s8juy');
		for (var header of list_header){
		 header.setAttribute("style", "opacity:0;position:absolute; top:-100px;");
		 };
		 
		var list_main = document.querySelectorAll('main');
		for (var main of list_main){
		 main.setAttribute("style", "padding:0px;");
		 };
		 
		var list_hoverable = document.querySelectorAll('.css-1qonaiz-panel-container-panel-transparent-container');
		for (var hoverable of list_hoverable){
		 hoverable.classList = [];
		 };

		var list_hidden_buttons = document.querySelectorAll('.css-jfq0f7-toolbar-button-panel-menu, .css-sdnmo1-hover-container-widget');
		for (var hidden_buttons of list_hidden_buttons){
		 hidden_buttons.remove();
		 };
	}
}

async function longLoop () {
	console.log("run makeCustom");
	makeCustom();
 
	setTimeout(longLoop, 500);
}
longLoop();
